/**
 * Declaring the function enxStart() sets a code block to run
 * once automatically when the app starts running. It's
 * used to perform setup tasks such as creating the
 * canvas and initializing variables
 * @returns {void}
 *
 */
function enxStart() {
  enxCreateScreen(1000, 1000);
  enxBackground(0, 0, 0);
  /** @type {int} */
  const cellWidth = Math.floor(enxWidth / GRID_COLS);
  /** @type {int} */
  const cellHeight = Math.floor(enxHeight / GRID_ROWS);
  for (let x = 0; x <= GRID_COLS; ++x) {
    enxStroke(0, 255, 0);
    enxLine(x * cellWidth, 0, x * cellWidth, ENX_CANVAS_CTX.canvas.height);
    for (let y = 0; y <= GRID_ROWS; ++y) {
      enxStroke(255, 255, 0);
      enxLine(0, y * cellHeight, enxWidth, y * cellHeight);
    }
  }

  let p1 = new enxVector(enxWidth * 0.43, enxHeight * 0.33);
  enxFill(255, 0, 255);
  enxNoStroke();
  enxCircle(...p1.array(), 0.2 * cellWidth);

  let p2 = new enxVector(enxWidth * 0.33, enxHeight * 0.43);
  enxFill(255, 0, 255);
  enxNoStroke();
  enxCircle(...p2.array(), 0.2 * cellWidth);

  enxStroke(255, 0, 255);
  enxLine(...p1.array(), ...p2.array());

  //   enxScreen.addEventListener("mousemove", (event) => {});

  console.log("WORKNG");
}

enxStart();
